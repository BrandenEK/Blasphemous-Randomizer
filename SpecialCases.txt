
Randomizer - OnSceneLoaded()
{
	if (scene is holy visage room) x 3
	{
		//Give visage reward
		Give visage reward;
		Set flag altar_done;
		Disable altar trigger;
	}
}

FlagModification - OnEnter()
{
	if (scene is albero sick house)
	{
		set if doesn't contain "_DEAD";
	}
}

FlagExists - OnEnter()
{
	if (scene is bridge fight) x 3
	{
		//Spawn fight if 3 wounds are owned
		flag is equal to ifQuestItemOwned;
	}
	if (scene is albero sick house)
	{
		//Only give last reward if all herbs delivered
		flag is equal to ifAllHerbsDelivered;
	}
	if (scene is gemino area)
	{
		//Make gemino never turn wooden
		flag is false if text is "_bossdead";
	}
	if (scene is redento corpse room)
	{
		//Remove duplicate redento item
		return if get flag "LOCATION_" + id;
	}
	if (scene is altasgracias room)
	{
		//Remove duplicate altasgracias item
		return if get flag "LOCATION_" + id;
	}
	if (scene is gemino area or gemino cave)
	{
		//Remove frozen olive duplicate
		return if get flag "LOCATION_" + id;
	}
	if (scene is gemino area or gemino cave)
	{
		//Remove dried flowers duplicate
		return if get flag "LOCATION_" + id;
	}
}

ItemIsOwned - executeAction()
{
	if (scene is first 3 boss room) x 3
	{
		//Remove barricades if collected item
		return if get flag "LOCATION_" + id;
	}
	if (scene is mask room) x 3
	{
		//Leave room if collected item
		return if get flag "LOCATION_" + id;
	}
	if (scene is albero graveyard)
	{
		//Never spawn graveyard item
		return true;
	}
	if (scene is albero sick house)
	{
		//Allow first reward to be collected
		return false;
	}
	if (scene is bridge fight)
	{
		//Remove duplicate Esdras item
		return if get flag "LOCATION_" + id;
	}
	if (scene is isidora fight)
	{
		//Remove duplicate isidora item
		return if get flag "LOCATION_" + id;
	}
	if (scene is redento room)
	{
		//Remove duplicate redento corpse item
		return if get flag "LOCATION_" + id;
	}
	if (scene is gemino area)
	{
		//Remove thimble duplicate if not talked to before boss)
		return if get flag "LOCATION_" + id;
	}
	if (scene is gemino area or gemino cave)
	{
		//Remove frozen olive duplicate
		return if get flag "LOCATION_" + id;
	}
}
